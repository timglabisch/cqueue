CREATE KEYSPACE queue WITH REPLICATION = { 'class' : 'SimpleStrategy', 'replication_factor' : 1 };

use queue;

create table queue_locks (
	queue text,
	part int,
	lock timestamp,
	owner text,
	PRIMARY KEY(queue, part)
);

create table queue (
 queue text,
 part int,
 id int,
 msg text,
 date timestamp,
 PRIMARY KEY((queue, part), id)
);
